
# VHDL

- use only end entity;

    entity X
    end entity;

- use only end architecture;

    architecture Y of X
    end architecture;

# Coding Style

g - gate, prefix, e.g. g1a0 -- gate 1, input a0

_n - active low, suffix, clk_n 

d - data input
q - data output

clk - clock
rst - reset

_i ? internal?
